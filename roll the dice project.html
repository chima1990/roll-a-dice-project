// Import readline module for user input const readline = require('readline'); // Create an interface for user input const rl = readline.createInterface({ input: process.stdin, output: process.stdout }); // Function to roll a dice function rollDice() { // Prompt the user to press Enter to roll the dice rl.question("Press Enter to roll the dice...", () => { // Generate a random number between 1 and 6 const diceRoll = Math.floor(Math.random() * 6) + 1; // Display the value rolled to the user console.log(`You rolled a ${diceRoll}!`); // Close the readline interface rl.close(); }); } // Run the dice roll program rollDice();
